	s_address
from
	supplier,
	nation

	s_suppkey in (
		select
			ps_suppkey

			partsupp

			ps_partkey in (
				select
					p_partkey
				from
					part

					p_name like 'almond%'
			)
			and ps_availqty > (
				select
					0.5 * sum(l_quantity)
				from
					lineitem
				where
					l_partkey = ps_partkey
					and l_suppkey = ps_suppkey
					and l_shipdate >= '1996-01-01'
					and l_shipdate < cast(date '1996-01-01' + interval '1 year' as date)
			)
	)
	and s_nationkey = n_nationkey
	and n_name = 'ETHIOPIA'
order by
	s_name;
